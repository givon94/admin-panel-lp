# Пример файла robots.txt для настройки гипотетического сайта https://ДОМЕН.by/

# Указываем правильный вид сайта (для сайтов с SSL сертификатом, указываем https:// протокол)
Host: https://ДОМЕН.by/

# Указываем где размещается карта сайта (файл sitemap.xml) 
Sitemap: https://ДОМЕН.by/sitemap.xml

# Правила для всех ботов
User-agent: *
Allow: /css/*.css
Disallow: /cgi-bin/

Disallow: /js/

Disallow: *utm=
